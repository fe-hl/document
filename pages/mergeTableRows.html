interface IMergeTableRows<T = any> {
  rowIndex: number; // 行
  columnIndex: number; // 列
  list: T[]; // 列表数据
  rules: Record<string, number | Array<number>>; // 合并规则 {id:0} OR {id:[2,4]} OR {'id~name': [0, 1]}
}
